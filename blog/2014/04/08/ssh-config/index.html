SSH Setting

tunnel

config file

#vi ~/.ssh/config
Host *
    StrictHostKeyChecking no
    UserKnownHostsFile /dev/null
    LogLevel=quiet 
    ForwardX11=yes

Host vps
    HostName 23.252.110.14
    Port 22
    User www-data
    IdentityFile ~/.ssh/id_rsa
